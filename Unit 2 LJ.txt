def Print_Volume(r):                    # Defining of function with 'r' parameter.
    pi = 3.14                           # Approximate of pi for calculation.
    float(r)                            # Turning integer to float type.
    print(round(4.0/3.0*pi*r**3,2))     # Calculation for sphere volume -and rounding the output to 2 decimal 					#    places.

print("The volume of a sphere with three different radiuses.")
print()
print("Radius of 3:")
Print_Volume(3)                         # Calling function with different radiuses.
print()
print("Radius of 4:")
Print_Volume(4)
print()
print("Radius of 5:")
Print_Volume(5)
